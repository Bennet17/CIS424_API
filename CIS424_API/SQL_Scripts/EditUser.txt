CREATE PROCEDURE [dbo].[sp_EditUser]
    @ID INT,
    @storeID INT,
    @username VARCHAR(50),
    @name VARCHAR(50),
    @position VARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE Users
    SET name = @name,
    storeID = @storeID,
    username = @username,
    position = @position
	WHERE ID = @ID;

END;
