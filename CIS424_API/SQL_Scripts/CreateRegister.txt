CREATE PROCEDURE [dbo].[sp_CreateRegister]
    @location VARCHAR(50),
    @ResultMessage VARCHAR(255) OUTPUT
AS
BEGIN
    SET NOCOUNT ON;
		--Get the storeID based on location entered by user
		DECLARE @storeID INT;
		SET @storeID = (SELECT ID FROM [dbo].[Store] WHERE location = @location);

        -- Insert the new register into the Registers table
        INSERT INTO Register (storeID)
        VALUES (@storeID);

        SET @ResultMessage = 'Register created successfully.';
END;
