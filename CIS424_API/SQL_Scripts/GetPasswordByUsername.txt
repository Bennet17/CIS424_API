CREATE PROCEDURE [dbo].[sp_GetPasswordByUsername]
    @username VARCHAR(255)
AS
BEGIN
    SET NOCOUNT ON;

    -- Retrieve the password for the given username
    SELECT hashPassword
    FROM Users
    WHERE username = @username;
END;
GO
